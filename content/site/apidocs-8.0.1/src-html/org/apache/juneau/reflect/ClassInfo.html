030/**
031 * Lightweight utility class for introspecting information about a class.
032 *
033 * <p>
034 * Provides various convenience methods for introspecting fields/methods/annotations
035 * that aren't provided by the standard Java reflection APIs.
036 *
037 * <p>
038 * Objects are designed to be lightweight to create and threadsafe.
039 *
040 * <h5 class='figure'>
041 * <p class='bpcode w800'>
042 *    <jc>// Wrap our class inside a ClassInfo.</jc>
043 *    ClassInfo ci = ClassInfo.<jsm>of</jsm>(MyClass.<jk>class</jk>);
044 *
045 *    <jc>// Get all methods in parent-to-child order, sorted alphabetically per class.</jc>
046 *    <jk>for</jk> (MethodInfo mi : ci.getAllMethodInfos(<jk>true</jk>, <jk>true</jk>)) {
047 *       <jc>// Do something with it.</jc>
048 *    }
049 *
050 *    <jc>// Get all class-level annotations in parent-to-child order.</jc>
051 *    <jk>for</jk> (MyAnnotation a : ci.getAnnotations(MyAnnotation.<jk>class</jk>, <jk>true</jk>)) {
052 *       // Do something with it.
053 *    }
054 * </p>
055 */
056@BeanIgnore

062   private List<ClassInfo> interfaces, declaredInterfaces, parents, allParents;

064   private List<ConstructorInfo> publicConstructors, declaredConstructors;

066   private Map<Class<?>,Optional<Annotation>> annotationMap, declaredAnnotationMap;

198   /**
199    * Returns a list of interfaces defined on this class and superclasses.
200    *
201    * @return
202    *    An unmodifiable list of interfaces defined on this class and superclasses.
203    *    <br>Results are in child-to-parent order.
204    */
205   public List<ClassInfo> getInterfaces() {
206      if (interfaces == null) {
207         Set<ClassInfo> s = new LinkedHashSet<>();
208         for (ClassInfo ci : getParents())
209            for (ClassInfo ci2 : ci.getDeclaredInterfaces()) {
210               s.add(ci2);
211               for (ClassInfo ci3 : ci2.getInterfaces())
212                  s.add(ci3);
213            }
214         interfaces = unmodifiableList(new ArrayList<>(s));
215      }
216      return interfaces;
217   }

434                  && m.hasArgs(String.class)

436               return m;
437      return null;

461               }
462            }
463         }
464      }
465      return null;
466   }

625   /**
626    * Finds a constructor with the specified parameters without throwing an exception.
627    *
628    * @param vis The minimum visibility.
629    * @param argTypes
630    *    The argument types in the constructor.
631    *    Can be subtypes of the actual constructor argument types.
632    * @return The matching constructor, or <jk>null</jk> if constructor could not be found.
633    */

635      return getConstructor(vis, false, argTypes);
636   }

638   private ConstructorInfo getConstructor(Visibility vis, boolean fuzzyArgs, Class<?>...argTypes) {
639      if (fuzzyArgs) {
640         int bestCount = -1;
641         ConstructorInfo bestMatch = null;
642         for (ConstructorInfo n : getDeclaredConstructors()) {
643            if (vis.isVisible(n.inner())) {
644               int m = ClassUtils.fuzzyArgsMatch(n.getParamTypes(), argTypes);
645               if (m > bestCount) {
646                  bestCount = m;
647                  bestMatch = n;
648               }
649            }
650         }
651         return bestMatch;
652      }
653
654      boolean isMemberClass = isNonStaticMemberClass();
655      for (ConstructorInfo n : getDeclaredConstructors()) {
656         List<ClassInfo> paramTypes = n.getParamTypes();
657         if (isMemberClass)
658            paramTypes = paramTypes.subList(1, paramTypes.size());
659         if (ClassUtils.argsMatch(paramTypes, argTypes) && vis.isVisible(n.inner()))
660            return n;
661      }
662
663      return null;
664   }

670   /**
671    * Locates the no-arg constructor for this class.
672    *
673    * <p>
674    * Constructor must match the visibility requirements specified by parameter 'v'.
675    * If class is abstract, always returns <jk>null</jk>.
676    * Note that this also returns the 1-arg constructor for non-static member classes.
677    *
678    * @param v The minimum visibility.
679    * @return The constructor, or <jk>null</jk> if no no-arg constructor exists with the required visibility.
680    */
681   public ConstructorInfo getNoArgConstructor(Visibility v) {
682      if (isAbstract())
683         return null;
684      boolean isMemberClass = isNonStaticMemberClass();
685      for (ConstructorInfo cc : getDeclaredConstructors())
686         if (cc.hasNumParams(isMemberClass ? 1 : 0) && cc.isVisible(v))
687            return cc.makeAccessible(v);
688      return null;
689   }

1936   /**
1937    * Finds the real parameter type of this class.
1938    *
1939    * @param index The zero-based index of the parameter to resolve.
1940    * @param pt The parameterized type class containing the parameterized type to resolve (e.g. <c>HashMap</c>).
1941    * @return The resolved real class.
1942    */
1943   public Class<?> getParameterType(int index, Class<?> pt) {
1944      if (pt == null)
1945         throw new FormattedIllegalArgumentException("Parameterized type cannot be null");
1946
1947      // We need to make up a mapping of type names.
1948      Map<Type,Type> typeMap = new HashMap<>();
1949      Class<?> cc = c;
1950      while (pt != cc.getSuperclass()) {
1951         extractTypes(typeMap, cc);
1952         cc = cc.getSuperclass();
1953         if (cc == null)
1954            throw new FormattedIllegalArgumentException("Class ''{0}'' is not a subclass of parameterized type ''{1}''", c.getSimpleName(), pt.getSimpleName());
1955      }
1956
1957      Type gsc = cc.getGenericSuperclass();
1958
1959      if (! (gsc instanceof ParameterizedType))
1960         throw new FormattedIllegalArgumentException("Class ''{0}'' is not a parameterized type", pt.getSimpleName());
1961
1962      ParameterizedType cpt = (ParameterizedType)gsc;
1963      Type[] atArgs = cpt.getActualTypeArguments();
1964      if (index >= atArgs.length)
1965         throw new FormattedIllegalArgumentException("Invalid type index. index={0}, argsLength={1}", index, atArgs.length);
1966      Type actualType = cpt.getActualTypeArguments()[index];
1967
1968      if (typeMap.containsKey(actualType))
1969         actualType = typeMap.get(actualType);
1970
1971      if (actualType instanceof Class) {
1972         return (Class<?>)actualType;
1973
1974      } else if (actualType instanceof GenericArrayType) {
1975         Type gct = ((GenericArrayType)actualType).getGenericComponentType();
1976         if (gct instanceof ParameterizedType)
1977            return Array.newInstance((Class<?>)((ParameterizedType)gct).getRawType(), 0).getClass();
1978      } else if (actualType instanceof TypeVariable) {
1979         TypeVariable<?> typeVariable = (TypeVariable<?>)actualType;
1980         List<Class<?>> nestedOuterTypes = new LinkedList<>();
1981         for (Class<?> ec = cc.getEnclosingClass(); ec != null; ec = ec.getEnclosingClass()) {
1982            Class<?> outerClass = cc.getClass();
1983            nestedOuterTypes.add(outerClass);
1984            Map<Type,Type> outerTypeMap = new HashMap<>();
1985            extractTypes(outerTypeMap, outerClass);
1986            for (Map.Entry<Type,Type> entry : outerTypeMap.entrySet()) {
1987               Type key = entry.getKey(), value = entry.getValue();
1988               if (key instanceof TypeVariable) {
1989                  TypeVariable<?> keyType = (TypeVariable<?>)key;
1990                  if (keyType.getName().equals(typeVariable.getName()) && isInnerClass(keyType.getGenericDeclaration(), typeVariable.getGenericDeclaration())) {
1991                     if (value instanceof Class)
1992                        return (Class<?>)value;
1993                     typeVariable = (TypeVariable<?>)entry.getValue();
1994                  }
1995               }
1996            }
1997         }
1998      } else if (actualType instanceof ParameterizedType) {
1999         return (Class<?>)((ParameterizedType)actualType).getRawType();
2000      }
2001      throw new FormattedIllegalArgumentException("Could not resolve variable ''{0}'' to a type.", actualType.getTypeName());
2002   }

2004   private static boolean isInnerClass(GenericDeclaration od, GenericDeclaration id) {
2005      if (od instanceof Class && id instanceof Class) {
2006         Class<?> oc = (Class<?>)od;
2007         Class<?> ic = (Class<?>)id;
2008         while ((ic = ic.getEnclosingClass()) != null)
2009            if (ic == oc)
2010               return true;
2011      }
2012      return false;
2013   }

2015   private static void extractTypes(Map<Type,Type> typeMap, Class<?> c) {
2016      Type gs = c.getGenericSuperclass();
2017      if (gs instanceof ParameterizedType) {
2018         ParameterizedType pt = (ParameterizedType)gs;
2019         Type[] typeParameters = ((Class<?>)pt.getRawType()).getTypeParameters();
2020         Type[] actualTypeArguments = pt.getActualTypeArguments();
2021         for (int i = 0; i < typeParameters.length; i++) {
2022            if (typeMap.containsKey(actualTypeArguments[i]))
2023               actualTypeArguments[i] = typeMap.get(actualTypeArguments[i]);
2024            typeMap.put(typeParameters[i], actualTypeArguments[i]);
2025         }
2026      }
2027   }
