034/**
035 * Used to denote that a class is a REST resource and to associate metadata on it.
036 *
037 * <p>
038 * Usually used on a subclass of {@link RestServlet}, but can be used to annotate any class that you want to expose as
039 * a REST resource.
040 *
041 * <h5 class='section'>See Also:</h5>
042 * <ul>
043 *    <li class='link'>{@doc juneau-rest-server.RestResource}
044 * </ul>
045 */
046@Documented
047@Target(TYPE)
048@Retention(RUNTIME)
049@Inherited
050public @interface RestResource {
051
052   /**
053    * Allow body URL parameter.
054    *
055    * <p>
056    * When enabled, the HTTP body content on PUT and POST requests can be passed in as text using the <js>"body"</js>
057    * URL parameter.
058    * <br>
059    * For example:
060    * <p class='bcode w800'>
061    *  ?body=(name='John%20Smith',age=45)
062    * </p>
063    *
064    * <h5 class='section'>Notes:</h5>
065    * <ul class='spaced-list'>
066    *    <li>
067    *       Supports {@doc DefaultRestSvlVariables}
068    *       (e.g. <js>"$L{my.localized.variable}"</js>).
069    * </ul>
070    *
071    * <h5 class='section'>See Also:</h5>
072    * <ul>
073    *    <li class='jf'>{@link RestContext#REST_allowBodyParam}
074    * </ul>
075    */
076   String allowBodyParam() default "";
077
078   /**
079    * Allowed method parameters.
080    *
081    * <p>
082    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular
083    * GET request.
084    * <br>
085    * For example:
086    * <p class='bcode w800'>
087    *  ?method=OPTIONS
088    * </p>
089    *
090    * <h5 class='section'>Notes:</h5>
091    * <ul class='spaced-list'>
092    *    <li>
093    *       Supports {@doc DefaultRestSvlVariables}
094    *       (e.g. <js>"$L{my.localized.variable}"</js>).
095    * </ul>
096    *
097    * <h5 class='section'>See Also:</h5>
098    * <ul>
099    *    <li class='jf'>{@link RestContext#REST_allowedMethodParams}
100    * </ul>
101    */
102   String allowedMethodParams() default "";
103
104   /**
105    * Allow header URL parameters.
106    *
107    * <p>
108    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query
109    * parameters.
110    * <br>
111    * For example:
112    * <p class='bcode w800'>
113    *  ?Accept=text/json&amp;Content-Type=text/json
114    * </p>
115    *
116    * <h5 class='section'>Notes:</h5>
117    * <ul class='spaced-list'>
118    *    <li>
119    *       Supports {@doc DefaultRestSvlVariables}
120    *       (e.g. <js>"$L{my.localized.variable}"</js>).
121    * </ul>
122    *
123    * <h5 class='section'>See Also:</h5>
124    * <ul>
125    *    <li class='jf'>{@link RestContext#REST_allowHeaderParams}
126    * </ul>
127    */
128   String allowHeaderParams() default "";
129
130   /**
131    * Class-level bean filters.
132    *
133    * <p>
134    * Shortcut to add bean filters to the bean contexts of the objects returned by the following methods:
135    * <ul>
136    *    <li>{@link RestContext#getBeanContext()}
137    *    <li>{@link RestContext#getSerializers()}
138    *    <li>{@link RestContext#getParsers()}
139    * </ul>
140    *
141    * <h5 class='section'>See Also:</h5>
142    * <ul>
143    *    <li class='jf'>{@link BeanContext#BEAN_beanFilters}
144    * </ul>
145    *
146    * @deprecated Use {@link BeanConfig#beanFilters()}
147    */
148   @Deprecated
149   Class<?>[] beanFilters() default {};

170    * <h5 class='section'>See Also:</h5>

172    *    <li class='jf'>{@link RestContext#REST_children}

174    */

195    *
196    * <h5 class='section'>Notes:</h5>
197    * <ul class='spaced-list'>
198    *    <li>

200    *       (e.g. <js>"$L{my.localized.variable}"</js>).
201    * </ul>
202    *
203    * <h5 class='section'>See Also:</h5>
204    * <ul>
205    *    <li class='jf'>{@link RestContext#REST_clientVersionHeader}
206    * </ul>
207    */
208   String clientVersionHeader() default "";
209
210   /**
211    * Optional location of configuration file for this servlet.
212    *
213    * <p>
214    * The configuration file .
215    *
216    * <h5 class='section'>Notes:</h5>
217    * <ul class='spaced-list'>
218    *    <li>
219    *       Supports {@doc DefaultRestSvlVariables}
220    *       (e.g. <js>"$L{my.localized.variable}"</js>).
221    *    <li>
222    *       Use the keyword <code>SYSTEM_DEFAULT</code> to refer to the system default configuration
223    *       returned by the {@link Config#getSystemDefault()}.
224    * </ul>
225    *
226    * <h5 class='section'>See Also:</h5>
227    * <ul>
228    *    <li class='jm'>{@link RestContextBuilder#config(Config)}

232

235    *

239    * <h5 class='section'>See Also:</h5>
240    * <ul>

263

270    * <h5 class='section'>Notes:</h5>
271    * <ul class='spaced-list'>
272    *    <li>
273    *       Supports {@doc DefaultRestSvlVariables}

275    * </ul>
276    *
277    * <h5 class='section'>See Also:</h5>

283
284   /**

286    *
287    * <p>
288    * The default value for the <code>Content-Type</code> header if not specified on a request.
289    *
290    * <p>
291    * This is a shortcut for using {@link #defaultRequestHeaders()} for just this specific header.

293    * <h5 class='section'>Notes:</h5>

295    *    <li>

301
302   /**
303    * Default request headers.
304    *
305    * <p>

307    *
308    * <h5 class='section'>Notes:</h5>
309    * <ul class='spaced-list'>
310    *    <li>

315    * <h5 class='section'>See Also:</h5>
316    * <ul>
317    *    <li class='jf'>{@link RestContext#REST_defaultRequestHeaders}

321
322   /**
323    * Default response headers.
324    *
325    * <p>
326    * Specifies default values for response headers if they're not set after the Java REST method is called.
327    *
328    * <h5 class='section'>Notes:</h5>
329    * <ul class='spaced-list'>
330    *    <li>
331    *       Supports {@doc DefaultRestSvlVariables}
332    *       (e.g. <js>"$L{my.localized.variable}"</js>).
333    * </ul>
334    *
335    * <h5 class='section'>See Also:</h5>
336    * <ul>
337    *    <li class='jf'>{@link RestContext#REST_defaultResponseHeaders}
338    * </ul>
339    */
340   String[] defaultResponseHeaders() default {};
341
342   /**
343    * Optional servlet description.
344    *
345    * <p>
346    * It is used to populate the Swagger description field.
347    * <br>This value can be retrieved programmatically through the {@link RestRequest#getResourceDescription()} method.
348    *
349    * <h5 class='section'>Notes:</h5>
350    * <ul class='spaced-list'>
351    *    <li>
352    *       Supports {@doc DefaultRestSvlVariables}
353    *       (e.g. <js>"$L{my.localized.variable}"</js>).
354    *    <li>
355    *       The format is plain-text.
356    *       <br>Multiple lines are concatenated with newlines.
357    * </ul>
358    *
359    * <h5 class='section'>See Also:</h5>
360    * <ul>
361    *    <li class='jm'>{@link RestInfoProvider#getDescription(RestRequest)}
362    * </ul>
363    */
364   String[] description() default {};
365
366   /**
367    * Compression encoders.
368    *
369    * <p>
370    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
371    *
372    * <h5 class='section'>See Also:</h5>
373    * <ul>
374    *    <li class='jf'>{@link RestContext#REST_encoders}
375    * </ul>
376    */

378
379   /**
380    * Shortcut for setting {@link #properties()} of simple boolean types.
381    *
382    * <h5 class='section'>Notes:</h5>
383    * <ul class='spaced-list'>
384    *    <li>
385    *       Supports {@doc DefaultRestSvlVariables}
386    *       (e.g. <js>"$L{my.localized.variable}"</js>).
387    *    <li>
388    *       Setting a flag is equivalent to setting the same property to <js>"true"</js>.
389    * </ul>
390    */

420    *          <jc>// Widgets for $W variables.</jc>
421    *          widgets={
422    *             PoweredByJuneau.<jk>class</jk>,
423    *             ContentTypeLinks.<jk>class</jk>
