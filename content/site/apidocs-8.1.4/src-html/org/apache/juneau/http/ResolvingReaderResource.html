013package org.apache.juneau.http;
014
015import static org.apache.juneau.internal.IOUtils.*;
016
017import java.io.*;
018import java.util.*;
019
020import org.apache.juneau.http.annotation.*;
021import org.apache.juneau.svl.*;
022
023/**
024 * An extension of {@link ReaderResource} that allows automatic resolution of SVL variables.
025 */
026public class ResolvingReaderResource extends ReaderResource {
027
028   private final VarResolverSession varSession;
029
030   /**
031    * Constructor.
032    *
033    * @param b Builder containing values to initialize this object with.
034    * @throws IOException Thrown by underlying stream.
035    */
036   protected ResolvingReaderResource(ResolvingResourceReaderBuilder b) throws IOException {
037      super(b);
038      this.varSession = b.varResolver;
039   }
040
041   /**
042    * Constructor.
043    *
044    * @param mediaType The resource media type.
045    * @param headers The HTTP response headers for this streamed resource.
046    * @param varSession Optional variable resolver for resolving variables in the string.
047    * @param cached
048    *    Identifies if this resource is cached in memory.
049    *    <br>If <jk>true</jk>, the contents will be loaded into a String for fast retrieval.
050    * @param contents
051    *    The resource contents.
052    *    <br>If multiple contents are specified, the results will be concatenated.
053    *    <br>Contents can be any of the following:
054    *    <ul>
055    *       <li><c>InputStream</c>
056    *       <li><c>Reader</c> - Converted to UTF-8 bytes.
057    *       <li><c>File</c>
058    *       <li><c>CharSequence</c> - Converted to UTF-8 bytes.
059    *    </ul>
060    * @throws IOException Thrown by underlying stream.
061    */
062   public ResolvingReaderResource(MediaType mediaType, Map<String,Object> headers, boolean cached, VarResolverSession varSession, Object...contents) throws IOException {
063      super(mediaType, headers, cached, contents);
064      this.varSession = varSession;
065   }
066
067   //-----------------------------------------------------------------------------------------------------------------
068   // Builder
069   //-----------------------------------------------------------------------------------------------------------------
070
071   /**
072    * Creates a new instance of a {@link ResolvingResourceReaderBuilder} for this class.
073    *
074    * @return A new instance of a {@link ResolvingResourceReaderBuilder}.
075    */
076   public static ResolvingResourceReaderBuilder create() {
077      return new ResolvingResourceReaderBuilder();
078   }
079
080   @ResponseBody
081   @Override /* Writeable */
082   public Writer writeTo(Writer w) throws IOException {
083      for (Object o : contents) {
084         if (o != null) {
085            if (varSession == null)
086               pipe(o, w);
087            else
088               varSession.resolveTo(read(o), w);
089         }
090      }
091      return w;
092   }
093}
