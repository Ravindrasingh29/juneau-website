013package org.apache.juneau.http;
014
015import java.io.*;
016import java.util.*;
017
018import org.apache.juneau.internal.*;
019
020/**
021 * Builder class for constructing {@link StreamResource} objects.
022 *
023 * <ul class='seealso'>
024 *    <li class='link'>{@doc juneau-rest-server.RestMethod.StreamResource}
025 * </ul>
026 */
027public class StreamResourceBuilder {
028   ArrayList<Object> contents = new ArrayList<>();
029   MediaType mediaType;
030   Map<String,Object> headers = new LinkedHashMap<>();
031   boolean cached;
032
033   /**
034    * Specifies the resource media type string.
035    *
036    * @param mediaType The resource media type string.
037    * @return This object (for method chaining).
038    */
039   @FluentSetter
040   public StreamResourceBuilder mediaType(String mediaType) {
041      this.mediaType = MediaType.forString(mediaType);
042      return this;
043   }
044
045   /**
046    * Specifies the resource media type string.
047    *
048    * @param mediaType The resource media type string.
049    * @return This object (for method chaining).
050    */
051   @FluentSetter
052   public StreamResourceBuilder mediaType(MediaType mediaType) {
053      this.mediaType = mediaType;
054      return this;
055   }
056
057   /**
058    * Specifies the contents for this resource.
059    *
060    * <p>
061    * This method can be called multiple times to add more content.
062    *
063    * @param contents
064    *    The resource contents.
065    *    <br>If multiple contents are specified, the results will be concatenated.
066    *    <br>Contents can be any of the following:
067    *    <ul>
068    *       <li><code><jk>byte</jk>[]</code>
069    *       <li><c>InputStream</c>
070    *       <li><c>Reader</c> - Converted to UTF-8 bytes.
071    *       <li><c>File</c>
072    *       <li><c>CharSequence</c> - Converted to UTF-8 bytes.
073    *    </ul>
074    * @return This object (for method chaining).
075    */
076   @FluentSetter
077   public StreamResourceBuilder contents(Object...contents) {
078      Collections.addAll(this.contents, contents);
079      return this;
080   }
081
082   /**
083    * Specifies an HTTP response header value.
084    *
085    * @param name The HTTP header name.
086    * @param value
087    *    The HTTP header value.
088    *    <br>Will be converted to a <c>String</c> using {@link Object#toString()}.
089    * @return This object (for method chaining).
090    */
091   @FluentSetter
092   public StreamResourceBuilder header(String name, Object value) {
093      this.headers.put(name, value);
094      return this;
095   }
096
097   /**
098    * Specifies HTTP response header values.
099    *
100    * @param headers
101    *    The HTTP headers.
102    *    <br>Values will be converted to <c>Strings</c> using {@link Object#toString()}.
103    * @return This object (for method chaining).
104    */
105   @FluentSetter
106   public StreamResourceBuilder headers(Map<String,Object> headers) {
107      this.headers.putAll(headers);
108      return this;
109   }
110
111   /**
112    * Specifies HTTP response header values.
113    *
114    * @param headers
115    *    The HTTP headers.
116    *    <br>Values will be converted to <c>Strings</c> using {@link Object#toString()}.
117    * @return This object (for method chaining).
118    */
119   @FluentSetter
120   public StreamResourceBuilder headers(org.apache.http.Header...headers) {
121      for (org.apache.http.Header h : headers)
122         this.headers.put(h.getName(), h.getValue());
123      return this;
124   }
125
126   /**
127    * Specifies that this resource is intended to be cached.
128    *
129    * <p>
130    * This will trigger the contents to be loaded into a byte array for fast serializing.
131    *
132    * @return This object (for method chaining).
133    */
134   @FluentSetter
135   public StreamResourceBuilder cached() {
136      this.cached = true;
137      return this;
138   }
139
140   /**
141    * Create a new {@link StreamResource} using values in this builder.
142    *
143    * @return A new immutable {@link StreamResource} object.
144    * @throws IOException Thrown by underlying stream.
145    */
146   public StreamResource build() throws IOException {
147      return new StreamResource(this);
148   }
149
150   // <FluentSetters>
151
152   // </FluentSetters>
153}
