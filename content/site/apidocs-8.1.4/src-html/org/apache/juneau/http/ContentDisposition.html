013package org.apache.juneau.http;
014
015import org.apache.juneau.http.annotation.*;
016
017/**
018 * Represents a parsed <l>Content-Disposition</l> HTTP request header.
019 *
020 * <p>
021 * In a regular HTTP response, the Content-Disposition response header is a header indicating if the content is expected
022 * to be displayed inline in the browser, that is, as a Web page or as part of a Web page, or as an attachment, that is
023 * downloaded and saved locally.
024 *
025 * <h5 class='figure'>Example</h5>
026 * <p class='bcode w800'>
027 *    Content-Disposition: form-data; name="fieldName"; filename="filename.jpg"
028 * </p>
029 *
030 * <h5 class='topic'>RFC2616 Specification</h5>
031 *
032 * The Expect request-header field is used to indicate that particular server behaviors are required by the client.
033 * <p class='bcode w800'>
034 * content-disposition = "Content-Disposition" ":"
035 *    disposition-type *( ";" disposition-parm )
036 *    disposition-type = "attachment" | disp-extension-token
037 *    disposition-parm = filename-parm | disp-extension-parm
038 *    filename-parm = "filename" "=" quoted-string
039 * disp-extension-token = token
040 *    disp-extension-parm = token "=" ( token | quoted-string )
041 * </p>
042 *
043 * <ul class='seealso'>
044 *    <li class='extlink'>{@doc RFC2616}
045 * </ul>
046 */
047@Header("Content-Disposition")
048public final class ContentDisposition extends BasicStringHeader {
049
050   /**
051    * Returns a parsed <c>Content-Disposition</c> header.
052    *
053    * @param value The <c>Content-Disposition</c> header string.
054    * @return The parsed <c>Content-Disposition</c> header, or <jk>null</jk> if the string was null.
055    */
056   public static ContentDisposition forString(String value) {
057      if (value == null)
058         return null;
059      return new ContentDisposition(value);
060   }
061
062   /**
063    * Constructor.
064    *
065    * @param value The value for this header.
066    */
067   public ContentDisposition(String value) {
068      super("Expect", value);
069   }
070}
