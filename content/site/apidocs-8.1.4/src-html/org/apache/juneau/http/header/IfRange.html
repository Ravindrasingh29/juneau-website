013package org.apache.juneau.http.header;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.util.function.*;
018
019import org.apache.juneau.http.*;
020import org.apache.juneau.http.annotation.*;
021import org.apache.juneau.internal.*;
022
023/**
024 * Represents a parsed <l>If-Range</l> HTTP request header.
025 *
026 * <p>
027 * If the entity is unchanged, send me the part(s) that I am missing; otherwise, send me the entire new entity.
028 *
029 * <h5 class='figure'>Example</h5>
030 * <p class='bcode w800'>
031 *    If-Range: "737060cd8c284d8af7ad3082f209582d"
032 * </p>
033 *
034 * <h5 class='topic'>RFC2616 Specification</h5>
035 *
036 * If a client has a partial copy of an entity in its cache, and wishes to have an up-to-date copy of the entire entity
037 * in its cache, it could use the Range request-header with a conditional GET (using either or both of
038 * If-Unmodified-Since and If-Match.)
039 * However, if the condition fails because the entity has been modified, the client would then have to make a second
040 * request to obtain the entire current entity-body.
041 *
042 * <p>
043 * The If-Range header allows a client to "short-circuit" the second request.
044 * Informally, its meaning is `if the entity is unchanged, send me the part(s) that I am missing; otherwise, send me
045 * the entire new entity'.
046 * <p class='bcode w800'>
047 *    If-Range = "If-Range" ":" ( entity-tag | HTTP-date )
048 * </p>
049 *
050 * <p>
051 * If the client has no entity tag for an entity, but does have a Last- Modified date, it MAY use that date in an
052 * If-Range header.
053 * (The server can distinguish between a valid HTTP-date and any form of entity-tag by examining no more than two
054 * characters.)
055 * The If-Range header SHOULD only be used together with a Range header, and MUST be ignored if the request does not
056 * include a Range header, or if the server does not support the sub-range operation.
057 *
058 * <p>
059 * If the entity tag given in the If-Range header matches the current entity tag for the entity, then the server SHOULD
060 * provide the specified sub-range of the entity using a 206 (Partial content) response.
061 * If the entity tag does not match, then the server SHOULD return the entire entity using a 200 (OK) response.
062 *
063 * <ul class='seealso'>
064 *    <li class='extlink'>{@doc RFC2616}
065 * </ul>
066 */
067@Header("If-Range")
068public class IfRange extends BasicStringHeader {
069
070   private static final long serialVersionUID = 1L;
071
072   /**
073    * Convenience creator.
074    *
075    * @param value
076    *    The parameter value.
077    *    <br>Can be any of the following:
078    *    <ul>
079    *       <li>{@link String}
080    *       <li>Anything else - Converted to <c>String</c> then parsed.
081    *    </ul>
082    * @return A new {@link IfRange} object.
083    */
084   public static IfRange of(Object value) {
085      if (value == null)
086         return null;
087      return new IfRange(value);
088   }
089
090   /**
091    * Convenience creator using supplier.
092    *
093    * <p>
094    * Header value is re-evaluated on each call to {@link #getValue()}.
095    *
096    * @param value
097    *    The parameter value supplier.
098    *    <br>Can be any of the following:
099    *    <ul>
100    *       <li>{@link String}
101    *       <li>Anything else - Converted to <c>String</c> then parsed.
102    *    </ul>
103    * @return A new {@link IfRange} object.
104    */
105   public static IfRange of(Supplier<?> value) {
106      if (value == null)
107         return null;
108      return new IfRange(value);
109   }
110
111   /**
112    * Constructor.
113    *
114    * @param value
115    *    The parameter value.
116    *    <br>Can be any of the following:
117    *    <ul>
118    *       <li>{@link String}
119    *       <li>Anything else - Converted to <c>String</c> then parsed.
120    *       <li>A {@link Supplier} of anything on this list.
121    *    </ul>
122    */
123   public IfRange(Object value) {
124      super("If-Range", value);
125   }
126
127   /**
128    * Constructor.
129    *
130    * @param value
131    *    The parameter value.
132    */
133   public IfRange(String value) {
134      super("If-Range", value);
135   }
136
137   /**
138    * Returns this header value as a {@link java.util.Date} object.
139    *
140    * @return This header value as a {@link java.util.Date} object, or <jk>null</jk> if the value is not a date.
141    */
142   public java.util.Date asDate() {
143      char c0 = charAt(getValue(), 0), c1 = charAt(getValue(), 1);
144      if (c0 == '*' || c0 == '"' || (c0 == 'W' && c1 == '/'))
145         return null;
146      return DateUtils.parseDate(getValue());
147   }
148
149   /**
150    * Returns this header value as an {@link EntityValidator} object.
151    *
152    * @return
153    *    This header value as a {@link EntityValidator} object, or <jk>null</jk> if the value is not an entity
154    *    validator.
155    */
156   public EntityValidator asValidator() {
157      char c0 = charAt(getValue(), 0), c1 = charAt(getValue(), 1);
158      if (c0 == '*' || c0 == '"' || (c0 == 'W' && c1 == '/'))
159         return new EntityValidator(getValue());
160      return null;
161   }
162}
