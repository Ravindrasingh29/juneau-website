013package org.apache.juneau.http;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.util.function.*;
018
019import org.apache.http.*;
020import org.apache.juneau.*;
021import org.apache.juneau.http.header.*;
022import org.apache.juneau.httppart.*;
023import org.apache.juneau.oapi.*;
024import org.apache.juneau.serializer.*;
025import org.apache.juneau.urlencoding.*;
026
027/**
028 * Subclass of {@link NameValuePair} for serializing POJOs as URL-encoded form post entries using the
029 * {@link UrlEncodingSerializer class}.
030 *
031 * <h5 class='section'>Example:</h5>
032 * <p class='bcode w800'>
033 *    NameValuePairs params = <jk>new</jk> NameValuePairs()
034 *       .append(<jk>new</jk> SerializedNameValuePair(<js>"myPojo"</js>, pojo, UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>))
035 *       .append(<jk>new</jk> BasicNameValuePair(<js>"someOtherParam"</js>, <js>"foobar"</js>));
036 *    request.setEntity(<jk>new</jk> UrlEncodedFormEntity(params));
037 * </p>
038 */
039public class SerializedHeader extends BasicStringHeader implements NameValuePairable {
040   private static final long serialVersionUID = 1L;
041
042   private Object value;
043   private HttpPartSerializerSession serializer;
044   private HttpPartSchema schema;
045   private boolean skipIfEmpty;
046
047   /**
048    * Instantiates a new builder for this object.
049    *
050    * @return A new builder for this object.
051    */
052   public static SerializedHeaderBuilder create() {
053      return new SerializedHeaderBuilder();
054   }
055
056   /**
057    * Constructor.
058    *
059    * @param name The HTTP header name name.
060    * @param value The POJO to serialize to the parameter value.
061    * @param serializer
062    *    The serializer to use for serializing the value to a string value.
063    * @param schema
064    *    The schema object that defines the format of the output.
065    *    <br>If <jk>null</jk>, defaults to the schema defined on the serializer.
066    *    <br>If that's also <jk>null</jk>, defaults to {@link HttpPartSchema#DEFAULT}.
067    *    <br>Only used if serializer is schema-aware (e.g. {@link OpenApiSerializer}).
068    *    <br>Can also be a {@link Supplier}.
069    * @param skipIfEmpty If value is a blank string, the value should return as <jk>null</jk>.
070    */
071   public SerializedHeader(String name, Object value, HttpPartSerializerSession serializer, HttpPartSchema schema, boolean skipIfEmpty) {
072      super(name, null);
073      this.value = value;
074      this.serializer = serializer;
075      this.schema = schema == null ? HttpPartSchema.DEFAULT : schema;
076      this.skipIfEmpty = skipIfEmpty;
077   }
078
079   SerializedHeader(SerializedHeaderBuilder b) {
080      super(b.name, null);
081      this.value = b.value;
082      this.serializer = b.serializer;
083      this.schema = b.schema == null ? HttpPartSchema.DEFAULT : b.schema;
084   }
085
086   @Override /* NameValuePair */
087   public String getValue() {
088      try {
089         Object v = unwrap(value);
090         if (v == null) {
091            if (schema == null)
092               return null;
093            if (schema.getDefault() == null && ! schema.isRequired())
094               return null;
095         }
096         if (isEmpty(v) && skipIfEmpty && schema.getDefault() == null)
097            return null;
098         return serializer.serialize(HttpPartType.HEADER, schema, v);
099      } catch (SchemaValidationException e) {
100         throw new BasicRuntimeException(e, "Validation error on request {0} parameter ''{1}''=''{2}''", HttpPartType.HEADER, getName(), value);
101      } catch (SerializeException e) {
102         throw new BasicRuntimeException(e, "Serialization error on request {0} parameter ''{1}''", HttpPartType.HEADER, getName());
103      }
104   }
105
106   @Override /* NameValuePairable */
107   public NameValuePair asNameValuePair() {
108      return new SerializedNameValuePair(getName(), value, HttpPartType.HEADER, serializer, schema, skipIfEmpty);
109   }
110}
