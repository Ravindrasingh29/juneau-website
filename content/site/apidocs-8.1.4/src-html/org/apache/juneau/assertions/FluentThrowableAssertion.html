013package org.apache.juneau.assertions;
014
015import java.util.function.*;
016
017import org.apache.juneau.internal.*;
018
019/**
020 * Used for fluent assertion calls against throwables.
021 *
022 * @param <R> The return type.
023 */
024@FluentSetters(returns="FluentThrowableAssertion<R>")
025public class FluentThrowableAssertion<R> extends FluentAssertion<R> {
026
027   private final Throwable value;
028
029   /**
030    * Constructor.
031    *
032    * @param value The throwable being tested.
033    * @param returns The object to return after the test.
034    */
035   public FluentThrowableAssertion(Throwable value, R returns) {
036      super(returns);
037      this.value = value;
038   }
039
040
041   /**
042    * Asserts that this throwable is of the specified type.
043    *
044    * <h5 class='section'>Example:</h5>
045    * <p class='bcode w800'>
046    *    <jc>// Asserts that the specified method throws a RuntimeException. </jc>
047    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();})
048    *       .isType(RuntimeException.<jk>class</jk>);
049    * </p>
050    *
051    * @param type The type.
052    * @return This object (for method chaining).
053    */
054   public R isType(Class<?> type) {
055      if (! type.isInstance(value))
056         throw error("Exception was not expected type.\n\tExpected=[{0}]\n\tActual=[{1}]", className(type), className(value));
057      return returns();
058   }
059
060   /**
061    * Asserts that this throwable or any parent throwables contains all of the specified substrings.
062    *
063    * <h5 class='section'>Example:</h5>
064    * <p class='bcode w800'>
065    *    <jc>// Asserts that the specified method throws an exception with 'foobar' somewhere in the messages. </jc>
066    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).contains(<js>"foobar"</js>);
067    * </p>
068    *
069    * @param substrings The substrings to check for.
070    * @return This object (for method chaining).
071    */
072   public R contains(String...substrings) {
073      exists();
074      for (String substring : substrings) {
075         Throwable e2 = value;
076         boolean found = false;
077         while (e2 != null && ! found) {
078            found |= StringUtils.contains(e2.getMessage(), substring);
079            e2 = e2.getCause();
080         }
081         if (! found) {
082            throw error("Exception message did not contain expected substring.\n\tSubstring=[{0}]\n\tText=[{1}]", substring, value.getMessage());
083         }
084      }
085      return returns();
086   }
087
088   /**
089    * Asserts that this throwable has the specified message.
090    *
091    * <h5 class='section'>Example:</h5>
092    * <p class='bcode w800'>
093    *    <jc>// Asserts that the specified method throws an exception with the message 'foobar'.</jc>
094    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).is(<js>"foobar"</js>);
095    * </p>
096    *
097    * @param msg The message to check for.
098    * @return This object (for method chaining).
099    */
100   public R is(String msg) {
101      return message().is(msg);
102   }
103
104   /**
105    * Asserts that this throwable exists.
106    *
107    * <h5 class='section'>Example:</h5>
108    * <p class='bcode w800'>
109    *    <jc>// Asserts that the specified method throws any exception.</jc>
110    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).exists();
111    * </p>
112    *
113    * @return This object (for method chaining).
114    */
115   public R exists() {
116      if (value == null)
117         throw error("Exception was not thrown.");
118      return returns();
119   }
120
121   /**
122    * Asserts that this throwable doesn't exist.
123    *
124    * <h5 class='section'>Example:</h5>
125    * <p class='bcode w800'>
126    *    <jc>// Asserts that the specified method doesn't throw any exception.</jc>
127    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).notExists();
128    * </p>
129    *
130    * @return This object (for method chaining).
131    */
132   public R notExists() {
133      if (value != null)
134         throw error("Exception was thrown.");
135      return returns();
136   }
137
138   /**
139    * Asserts that the value passes the specified predicate test.
140    *
141    * @param test The predicate to use to test the value.
142    * @return The response object (for method chaining).
143    * @throws AssertionError If assertion failed.
144    */
145   public R passes(Predicate<Object> test) throws AssertionError {
146      if (! test.test(value))
147         throw error("Value did not pass predicate test.\n\tValue=[{0}]", value);
148      return returns();
149   }
150
151   /**
152    * Returns an assertion against the throwable message.
153    *
154    * <h5 class='section'>Example:</h5>
155    * <p class='bcode w800'>
156    *    <jc>// Asserts that the specified method throws an exception with 'foobar' somewhere in the messages. </jc>
157    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).message().matches(<js>".*foobar.*"</js>);
158    * </p>
159    *
160    * @return An assertion against the throwable message.  Never <jk>null</jk>.
161    */
162   public FluentStringAssertion<R> message() {
163      return new FluentStringAssertion<>(value == null ? null : value.getMessage(), returns());
164   }
165
166   /**
167    * Returns an assertion against the throwable localized message.
168    *
169    * <h5 class='section'>Example:</h5>
170    * <p class='bcode w800'>
171    *    <jc>// Asserts that the specified method throws an exception with 'foobar' somewhere in the localized messages. </jc>
172    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).localizedMessage().matches(<js>".*foobar.*"</js>);
173    * </p>
174    *
175    * @return An assertion against the throwable localized message.  Never <jk>null</jk>.
176    */
177   public FluentStringAssertion<R> localizedMessage() {
178      return new FluentStringAssertion<>(value == null ? null : value.getLocalizedMessage(), returns());
179   }
180
181   /**
182    * Returns an assertion against the throwable localized message.
183    *
184    * <h5 class='section'>Example:</h5>
185    * <p class='bcode w800'>
186    *    <jc>// Asserts that the specified method throws an exception with 'foobar' somewhere in the stack trace. </jc>
187    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).stackTrace().contains(<js>"foobar"</js>);
188    * </p>
189    *
190    * @return An assertion against the throwable stacktrace.  Never <jk>null</jk>.
191    */
192   public FluentStringAssertion<R> stackTrace() {
193      return new FluentStringAssertion<>(value == null ? null : StringUtils.getStackTrace(value), returns());
194   }
195
196   /**
197    * Returns an assertion against the caused-by throwable.
198    *
199    * <h5 class='section'>Example:</h5>
200    * <p class='bcode w800'>
201    *    <jc>// Asserts that the specified method throws an exception whose caused-by message contains 'foobar'. </jc>
202    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).causedBy().message().contains(<js>"foobar"</js>);
203    * </p>
204    *
205    * @return An assertion against the caused-by.  Never <jk>null</jk>.
206    */
207   public FluentThrowableAssertion<R> causedBy() {
208      return new FluentThrowableAssertion<>(value == null ? null : value.getCause(), returns());
209   }
210
211   /**
212    * Returns an assertion against the throwable localized message.
213    *
214    * <h5 class='section'>Example:</h5>
215    * <p class='bcode w800'>
216    *    <jc>// Asserts that the specified method throws an exception with a caused-by RuntimeException containing 'foobar'</jc>
217    *    ThrowableAssertion.<jsm>assertThrown</jsm>(() -&gt; {foo.getBar();}).causedBy(RuntimeException.<jk>class</jk>).exists().contains(<js>"foobar"</js>);
218    * </p>
219    *
220    * @param throwableClass The class type to search for in the caused-by chain.
221    * @return An assertion against the caused-by throwable.  Never <jk>null</jk>.
222    */
223   public FluentThrowableAssertion<R> find(Class<?> throwableClass) {
224      Throwable t = value;
225      while (t != null) {
226         if (throwableClass.isInstance(t))
227            return new FluentThrowableAssertion<>(t, returns());
228         t = t.getCause();
229      }
230      return new FluentThrowableAssertion<>(null, returns());
231   }
232
233   // <FluentSetters>
234
235   @Override /* GENERATED - Assertion */
236   public FluentThrowableAssertion<R> msg(String msg, Object...args) {
237      super.msg(msg, args);
238      return this;
239   }
240
241   @Override /* GENERATED - Assertion */
242   public FluentThrowableAssertion<R> stderr() {
243      super.stderr();
244      return this;
245   }
246
247   @Override /* GENERATED - Assertion */
248   public FluentThrowableAssertion<R> stdout() {
249      super.stdout();
250      return this;
251   }
252
253   // </FluentSetters>
254}
