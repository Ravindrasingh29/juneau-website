013package org.apache.juneau.assertions;
014
015import java.util.function.*;
016
017import org.apache.juneau.internal.*;
018import org.apache.juneau.json.*;
019import org.apache.juneau.reflect.*;
020import org.apache.juneau.serializer.*;
021
022/**
023 * Used for fluent assertion calls against POJOs.
024 *
025 * @param <R> The return type.
026 */
027@FluentSetters(returns="FluentObjectAssertion<R>")
028public class FluentObjectAssertion<R> extends FluentAssertion<R> {
029
030   private final Object value;
031
032   private static JsonSerializer JSON_BEANCOMPARE = JsonSerializer.create()
033      .ssq()
034      .keepNullProperties()
035      .addBeanTypes().addRootType()
036      .build();
037
038   private static JsonSerializer JSON_BEANCOMPARESORTED = JsonSerializer.create()
039      .ssq()
040      .sortCollections()
041      .sortMaps()
042      .keepNullProperties()
043      .addBeanTypes().addRootType()
044      .build();
045
046   private static JsonSerializer JSON = JsonSerializer.create()
047      .ssq()
048      .addBeanTypes().addRootType()
049      .build();
050
051   private static JsonSerializer JSON_SORTED = JsonSerializer.create()
052      .ssq()
053      .sortProperties()
054      .addBeanTypes().addRootType()
055      .build();
056
057   /**
058    * Constructor.
059    *
060    * @param o The object being tested.
061    * @param returns The object to return after the test.
062    */
063   public FluentObjectAssertion(Object o, R returns) {
064      super(returns);
065      this.value = o;
066   }
067
068   /**
069    * Asserts that the object is an instance of the specified class.
070    *
071    * <h5 class='section'>Example:</h5>
072    * <p class='bcode w800'>
073    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
074    *    <jsm>assertObject<jsm>(myPojo).instanceOf(MyBean.<jk>class</jk>);
075    * </p>
076    *
077    * @param parent The value to check against.
078    * @return The response object (for method chaining).
079    * @throws AssertionError If assertion failed.
080    */
081   public R isType(Class<?> parent) throws AssertionError {
082      if (value == null && parent == null)
083         return returns();
084      if (value == null && parent != null || value != null && parent == null || ! ClassInfo.of(value).isChildOf(parent))
085         throw error("Unexpected class.\n\tExpected=[{0}]\n\tActual=[{1}]", className(parent), className(value));
086      return returns();
087   }
088
089   /**
090    * Converts this object to text using the specified serializer and returns it as a new assertion.
091    *
092    * <h5 class='section'>Example:</h5>
093    * <p class='bcode w800'>
094    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
095    *    <jsm>assertObject<jsm>(myPojo).serialized(XmlSerializer.<jsf>DEFAULT</jsf>).is(<js>"&lt;object>&lt;foo>bar&lt;/foo>&lt;baz>qux&lt;/baz>&lt;/object>"</js>);
096    * </p>
097    *
098    * @param ws The serializer to use to convert the object to text.
099    * @return A new fluent string assertion.
100    */
101   public FluentStringAssertion<R> serialized(WriterSerializer ws) {
102      try {
103         String s = ws.serialize(this.value);
104         return new FluentStringAssertion<>(s, returns());
105      } catch (SerializeException e) {
106         throw new RuntimeException(e);
107      }
108   }
109
110   /**
111    * Converts this object to simplified JSON and returns it as a new assertion.
112    *
113    * <h5 class='section'>Example:</h5>
114    * <p class='bcode w800'>
115    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
116    *    <jsm>assertObject<jsm>(myPojo).json().is(<js>"{foo:'bar',baz:'qux'}"</js>);
117    * </p>
118    *
119    * @return A new fluent string assertion.
120    */
121   public FluentStringAssertion<R> json() {
122      return serialized(JSON);
123   }
124
125   /**
126    * Converts this object to sorted simplified JSON and returns it as a new assertion.
127    *
128    * <h5 class='section'>Example:</h5>
129    * <p class='bcode w800'>
130    *    <jc>// Validates that the specified object is an instance of MyBean.</jc>
131    *    <jsm>assertObject<jsm>(myPojo).jsonSorted().is(<js>"{baz:'qux',foo:'bar'}"</js>);
132    * </p>
133    *
134    * @return A new fluent string assertion.
135    */
136   public FluentStringAssertion<R> jsonSorted() {
137      return serialized(JSON_SORTED);
138   }
139
140   /**
141    * Verifies that two objects are equivalent after converting them both to JSON.
142    *
143    * @param o The object to compare against.
144    * @return The response object (for method chaining).
145    * @throws AssertionError If assertion failed.
146    */
147   public R sameAs(Object o) throws AssertionError {
148      return sameAsSerialized(o, JSON_BEANCOMPARE);
149   }
150
151   /**
152    * Verifies that two objects are equivalent after converting them both to sorted JSON.
153    *
154    * @param o The object to compare against.
155    * @return The response object (for method chaining).
156    * @throws AssertionError If assertion failed.
157    */
158   public R sameAsSorted(Object o) {
159      return sameAsSerialized(o, JSON_BEANCOMPARESORTED);
160   }
161
162   /**
163    * Asserts that the specified object is the same as this object after converting both to strings using the specified serializer.
164    *
165    * @param o The object to compare against.
166    * @param serializer The serializer to use to serialize this object.
167    * @return The response object (for method chaining).
168    * @throws AssertionError If assertion failed.
169    */
170   public R sameAsSerialized(Object o, WriterSerializer serializer) {
171      try {
172         String s1 = serializer.serialize(this.value);
173         String s2 = serializer.serialize(o);
174         if (! StringUtils.isEquals(s1, s2))
175            throw error("Unexpected comparison.\n\tExpected=[{0}]\n\tActual=[{1}]", s2, s1);
176      } catch (SerializeException e) {
177         throw new RuntimeException(e);
178      }
179      return returns();
180   }
181
182   /**
183    * Asserts that the value passes the specified predicate test.
184    *
185    * @param test The predicate to use to test the value.
186    * @return The response object (for method chaining).
187    * @throws AssertionError If assertion failed.
188    */
189   public R passes(Predicate<Object> test) throws AssertionError {
190      if (! test.test(value))
191         throw error("Value did not pass predicate test.\n\tValue=[{0}]", value);
192      return returns();
193   }
194
195   // <FluentSetters>
196
197   @Override /* GENERATED - Assertion */
198   public FluentObjectAssertion<R> msg(String msg, Object...args) {
199      super.msg(msg, args);
200      return this;
201   }
202
203   @Override /* GENERATED - Assertion */
204   public FluentObjectAssertion<R> stderr() {
205      super.stderr();
206      return this;
207   }
208
209   @Override /* GENERATED - Assertion */
210   public FluentObjectAssertion<R> stdout() {
211      super.stdout();
212      return this;
213   }
214
215   // </FluentSetters>
216}
