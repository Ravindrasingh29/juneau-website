013package org.apache.juneau.assertions;
014
015
016import java.util.function.*;
017
018import org.apache.juneau.internal.*;
019import org.apache.juneau.marshall.*;
020
021/**
022 * Used for fluent assertion calls against integers.
023 *
024 * <h5 class='section'>Example:</h5>
025 * <p class='bcode w800'>
026 *    <jc>// Validates the response status code is 200 or 404.</jc>
027 *    client
028 *       .get(<jsf>URL</jsf>)
029 *       .run()
030 *       .assertStatus().isAny(200,404);
031 * </p>
032 *
033 * @param <R> The return type.
034 */
035@FluentSetters(returns="FluentIntegerAssertion<R>")
036public class FluentIntegerAssertion<R> extends FluentAssertion<R> {
037
038   private final Integer value;
039
040   /**
041    * Constructor.
042    *
043    * @param value The value being tested.
044    * @param returns The object to return after the test.
045    */
046   public FluentIntegerAssertion(Integer value, R returns) {
047      super(returns);
048      this.value = value;
049   }
050
051   /**
052    * Asserts that the value equals the specified value.
053    *
054    * @param value The value to check against.
055    * @return The response object (for method chaining).
056    * @throws AssertionError If assertion failed.
057    */
058   public R equals(Integer value) throws AssertionError {
059      if (this.value == value)
060         return returns();
061      exists();
062      if (! this.value.equals(value))
063         throw error("Unexpected value.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
064      return returns();
065   }
066
067   /**
068    * Asserts that the value equals the specified value.
069    *
070    * <p>
071    * Equivalent to {@link #equals(Integer)}.
072    *
073    * @param value The value to check against.
074    * @return The response object (for method chaining).
075    * @throws AssertionError If assertion failed.
076    */
077   public R is(Integer value) throws AssertionError {
078      return equals(value);
079   }
080
081   /**
082    * Asserts that the value equals the specified value.
083    *
084    * @param value The value to check against.
085    * @return The response object (for method chaining).
086    * @throws AssertionError If assertion failed.
087    */
088   public R doesNotEqual(Integer value) throws AssertionError {
089      if (this.value != value)
090         return returns();
091      exists();
092      if (this.value.equals(value))
093         throw error("Unexpected value.\n\tExpected not=[{0}]\n\tActual=[{1}]", value, this.value);
094      return returns();
095   }
096
097   /**
098    * Asserts that the value equals the specified value.
099    *
100    * <p>
101    * Equivalent to {@link #doesNotEqual(Integer)}.
102    *
103    * @param value The value to check against.
104    * @return The response object (for method chaining).
105    * @throws AssertionError If assertion failed.
106    */
107   public R isNot(Integer value) throws AssertionError {
108      return doesNotEqual(value);
109   }
110
111   /**
112    * Asserts that the integer is not null.
113    *
114    * <p>
115    * Equivalent to {@link #isNotNull()}.
116    *
117    * @return The response object (for method chaining).
118    * @throws AssertionError If assertion failed.
119    */
120   public R exists() throws AssertionError {
121      return isNotNull();
122   }
123
124   /**
125    * Asserts that the integer is not null.
126    *
127    * <p>
128    * Equivalent to {@link #isNull()}.
129    *
130    * @return The response object (for method chaining).
131    * @throws AssertionError If assertion failed.
132    */
133   public R doesNotExist() throws AssertionError {
134      return isNull();
135   }
136
137   /**
138    * Asserts that the integer is not null.
139    *
140    * @return The response object (for method chaining).
141    * @throws AssertionError If assertion failed.
142    */
143   public R isNull() throws AssertionError {
144      if (value != null)
145         throw error("Value was not null.");
146      return returns();
147   }
148
149   /**
150    * Asserts that the integer is not null.
151    *
152    * @return The response object (for method chaining).
153    * @throws AssertionError If assertion failed.
154    */
155   public R isNotNull() throws AssertionError {
156      if (value == null)
157         throw error("Value was null.");
158      return returns();
159   }
160
161   /**
162    * Asserts that the value is one of the specified values.
163    *
164    * @param values The values to check against.
165    * @return The response object (for method chaining).
166    * @throws AssertionError If assertion failed.
167    */
168   public R isAny(Integer...values) throws AssertionError {
169      exists();
170      for (Integer v : values)
171         if (this.value.equals(v))
172            return returns();
173      throw error("Expected value not found.\n\tExpected=[{0}]\n\tActual=[{1}]", SimpleJson.DEFAULT.toString(values), value);
174   }
175
176   /**
177    * Asserts that the value is one of the specified values.
178    *
179    * @param values The values to check against.
180    * @return The response object (for method chaining).
181    * @throws AssertionError If assertion failed.
182    */
183   public R isNotAny(Integer...values) throws AssertionError {
184      exists();
185      for (Integer v : values)
186         if (this.value.equals(v))
187            throw error("Unexpected value found.\n\tUnexpected=[{0}]\n\tActual=[{1}]", v, value);
188      return returns();
189   }
190
191   /**
192    * Asserts that the value is greater than the specified value.
193    *
194    * @param value The values to check against.
195    * @return The response object (for method chaining).
196    * @throws AssertionError If assertion failed.
197    */
198   public R isGreaterThan(Integer value) throws AssertionError {
199      exists();
200      if (! (this.value > value))
201            throw error("Value was not greater than expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
202      return returns();
203   }
204
205   /**
206    * Asserts that the value is greater than the specified value.
207    *
208    * <p>
209    * Equivalent to {@link #isGreaterThan(Integer)}
210    *
211    * @param value The values to check against.
212    * @return The response object (for method chaining).
213    * @throws AssertionError If assertion failed.
214    */
215   public R isGt(Integer value) throws AssertionError {
216      return isGreaterThan(value);
217   }
218
219   /**
220    * Asserts that the value is greater than or equal to the specified value.
221    *
222    * @param value The values to check against.
223    * @return The response object (for method chaining).
224    * @throws AssertionError If assertion failed.
225    */
226   public R isGreaterThanOrEquals(Integer value) throws AssertionError {
227      exists();
228      if (! (this.value >= value))
229            throw error("Value was not greater than or equals to expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
230      return returns();
231   }
232
233   /**
234    * Asserts that the value is greater than or equal to the specified value.
235    *
236    * <p>
237    * Equivalent to {@link #isGreaterThanOrEquals(Integer)}
238    *
239    * @param value The values to check against.
240    * @return The response object (for method chaining).
241    * @throws AssertionError If assertion failed.
242    */
243   public R isGte(Integer value) throws AssertionError {
244      return isGreaterThanOrEquals(value);
245   }
246
247   /**
248    * Asserts that the value is less than the specified value.
249    *
250    * @param value The values to check against.
251    * @return The response object (for method chaining).
252    * @throws AssertionError If assertion failed.
253    */
254   public R isLessThan(Integer value) throws AssertionError {
255      exists();
256      if (! (this.value < value))
257            throw error("Value was not less than expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
258      return returns();
259   }
260
261   /**
262    * Asserts that the value is less than the specified value.
263    *
264    * <p>
265    * Equivalent to {@link #isLessThan(Integer)}
266    *
267    * @param value The values to check against.
268    * @return The response object (for method chaining).
269    * @throws AssertionError If assertion failed.
270    */
271   public R isLt(Integer value) throws AssertionError {
272      return isLessThan(value);
273   }
274
275   /**
276    * Asserts that the value is less than or equals to the specified value.
277    *
278    * @param value The values to check against.
279    * @return The response object (for method chaining).
280    * @throws AssertionError If assertion failed.
281    */
282   public R isLessThanOrEquals(Integer value) throws AssertionError {
283      exists();
284      if (! (this.value <= value))
285            throw error("Value was not less than or equals to expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
286      return returns();
287   }
288
289   /**
290    * Asserts that the value is less than or equals to the specified value.
291    *
292    * <p>
293    * Equivalent to {@link #isLessThanOrEquals(Integer)}
294    *
295    * @param value The values to check against.
296    * @return The response object (for method chaining).
297    * @throws AssertionError If assertion failed.
298    */
299   public R isLte(Integer value) throws AssertionError {
300      return isLessThanOrEquals(value);
301   }
302
303   /**
304    * Asserts that the value is between (inclusive) the specified upper and lower values.
305    *
306    * @param lower The lower value to check against.
307    * @param upper The upper value to check against.
308    * @return The response object (for method chaining).
309    * @throws AssertionError If assertion failed.
310    */
311   public R isBetween(Integer lower, Integer upper) throws AssertionError {
312      isLessThanOrEquals(upper);
313      isGreaterThanOrEquals(lower);
314      return returns();
315   }
316
317   /**
318    * Asserts that the value passes the specified predicate test.
319    *
320    * @param test The predicate to use to test the value.
321    * @return The response object (for method chaining).
322    * @throws AssertionError If assertion failed.
323    */
324   public R passes(Predicate<Integer> test) throws AssertionError {
325      if (! test.test(value))
326         throw error("Value did not pass predicate test.\n\tValue=[{0}]", value);
327      return returns();
328   }
329
330   // <FluentSetters>
331
332   @Override /* GENERATED - Assertion */
333   public FluentIntegerAssertion<R> msg(String msg, Object...args) {
334      super.msg(msg, args);
335      return this;
336   }
337
338   @Override /* GENERATED - Assertion */
339   public FluentIntegerAssertion<R> stderr() {
340      super.stderr();
341      return this;
342   }
343
344   @Override /* GENERATED - Assertion */
345   public FluentIntegerAssertion<R> stdout() {
346      super.stdout();
347      return this;
348   }
349
350   // </FluentSetters>
