013package org.apache.juneau.assertions;
014
015
016import java.util.function.*;
017
018import org.apache.juneau.internal.*;
019import org.apache.juneau.marshall.*;
020
021/**
022 * Used for fluent assertion calls against longs.
023 *
024 * <h5 class='section'>Example:</h5>
025 * <p class='bcode w800'>
026 *    <jc>// Validates the response length isn't too long.</jc>
027 *    client
028 *       .get(<jsf>URL</jsf>)
029 *       .run()
030 *       .assertLongHeader(<js>"Length"</js>).isLessThan(100000);
031 * </p>
032 *
033 * @param <R> The return type.
034 */
035@FluentSetters(returns="FluentLongAssertion<R>")
036public class FluentLongAssertion<R> extends FluentAssertion<R> {
037
038   private final Long value;
039
040   /**
041    * Constructor.
042    *
043    * @param value The value being tested.
044    * @param returns The object to return after the test.
045    */
046   public FluentLongAssertion(Long value, R returns) {
047      super(returns);
048      this.value = value;
049   }
050
051   /**
052    * Asserts that the value equals the specified value.
053    *
054    * @param value The value to check against.
055    * @return The response object (for method chaining).
056    * @throws AssertionError If assertion failed.
057    */
058   public R equals(Long value) throws AssertionError {
059      if (this.value == value)
060         return returns();
061      exists();
062      if (! this.value.equals(value))
063         throw error("Unexpected value.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
064      return returns();
065   }
066
067   /**
068    * Asserts that the value equals the specified value.
069    *
070    * <p>
071    * Equivalent to {@link #equals(Long)}.
072    *
073    * @param value The value to check against.
074    * @return The response object (for method chaining).
075    * @throws AssertionError If assertion failed.
076    */
077   public R is(Long value) throws AssertionError {
078      return equals(value);
079   }
080
081   /**
082    * Asserts that the value equals the specified value.
083    *
084    * <p>
085    * Equivalent to {@link #equals(Long)}.
086    *
087    * @param value The value to check against.
088    * @return The response object (for method chaining).
089    * @throws AssertionError If assertion failed.
090    */
091   public R is(int value) throws AssertionError {
092      return equals((long)value);
093   }
094
095   /**
096    * Asserts that the value equals the specified value.
097    *
098    * @param value The value to check against.
099    * @return The response object (for method chaining).
100    * @throws AssertionError If assertion failed.
101    */
102   public R doesNotEqual(Long value) throws AssertionError {
103      if (this.value != value)
104         return returns();
105      exists();
106      if (this.value.equals(value))
107         throw error("Unexpected value.\n\tExpected not=[{0}]\n\tActual=[{1}]", value, this.value);
108      return returns();
109   }
110
111   /**
112    * Asserts that the value equals the specified value.
113    *
114    * <p>
115    * Equivalent to {@link #doesNotEqual(Long)}.
116    *
117    * @param value The value to check against.
118    * @return The response object (for method chaining).
119    * @throws AssertionError If assertion failed.
120    */
121   public R isNot(Long value) throws AssertionError {
122      return doesNotEqual(value);
123   }
124
125   /**
126    * Asserts that the long is not null.
127    *
128    * <p>
129    * Equivalent to {@link #isNotNull()}.
130    *
131    * @return The response object (for method chaining).
132    * @throws AssertionError If assertion failed.
133    */
134   public R exists() throws AssertionError {
135      return isNotNull();
136   }
137
138   /**
139    * Asserts that the long is not null.
140    *
141    * <p>
142    * Equivalent to {@link #isNull()}.
143    *
144    * @return The response object (for method chaining).
145    * @throws AssertionError If assertion failed.
146    */
147   public R doesNotExist() throws AssertionError {
148      return isNull();
149   }
150
151   /**
152    * Asserts that the long is not null.
153    *
154    * @return The response object (for method chaining).
155    * @throws AssertionError If assertion failed.
156    */
157   public R isNull() throws AssertionError {
158      if (value != null)
159         throw error("Value was not null.");
160      return returns();
161   }
162
163   /**
164    * Asserts that the long is not null.
165    *
166    * @return The response object (for method chaining).
167    * @throws AssertionError If assertion failed.
168    */
169   public R isNotNull() throws AssertionError {
170      if (value == null)
171         throw error("Value was null.");
172      return returns();
173   }
174
175   /**
176    * Asserts that the value is one of the specified values.
177    *
178    * @param values The values to check against.
179    * @return The response object (for method chaining).
180    * @throws AssertionError If assertion failed.
181    */
182   public R isAny(Long...values) throws AssertionError {
183      exists();
184      for (Long v : values)
185         if (this.value.equals(v))
186            return returns();
187      throw error("Expected value not found.\n\tExpected=[{0}]\n\tActual=[{1}]", SimpleJson.DEFAULT.toString(values), value);
188   }
189
190   /**
191    * Asserts that the value is one of the specified values.
192    *
193    * @param values The values to check against.
194    * @return The response object (for method chaining).
195    * @throws AssertionError If assertion failed.
196    */
197   public R isNotAny(Long...values) throws AssertionError {
198      exists();
199      for (Long v : values)
200         if (this.value.equals(v))
201            throw error("Unexpected value found.\n\tUnexpected=[{0}]\n\tActual=[{1}]", v, value);
202      return returns();
203   }
204
205   /**
206    * Asserts that the value is greater than the specified value.
207    *
208    * @param value The values to check against.
209    * @return The response object (for method chaining).
210    * @throws AssertionError If assertion failed.
211    */
212   public R isGreaterThan(Long value) throws AssertionError {
213      exists();
214      if (! (this.value > value))
215            throw error("Value was not greater than expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
216      return returns();
217   }
218
219   /**
220    * Asserts that the value is greater than the specified value.
221    *
222    * <p>
223    * Equivalent to {@link #isGreaterThan(Long)}
224    *
225    * @param value The values to check against.
226    * @return The response object (for method chaining).
227    * @throws AssertionError If assertion failed.
228    */
229   public R isGt(Long value) throws AssertionError {
230      return isGreaterThan(value);
231   }
232
233   /**
234    * Asserts that the value is greater than or equal to the specified value.
235    *
236    * @param value The values to check against.
237    * @return The response object (for method chaining).
238    * @throws AssertionError If assertion failed.
239    */
240   public R isGreaterThanOrEquals(Long value) throws AssertionError {
241      exists();
242      if (! (this.value >= value))
243            throw error("Value was not greater than or equals to expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
244      return returns();
245   }
246
247   /**
248    * Asserts that the value is greater than or equal to the specified value.
249    *
250    * <p>
251    * Equivalent to {@link #isGreaterThanOrEquals(Long)}
252    *
253    * @param value The values to check against.
254    * @return The response object (for method chaining).
255    * @throws AssertionError If assertion failed.
256    */
257   public R isGte(Long value) throws AssertionError {
258      return isGreaterThanOrEquals(value);
259   }
260
261   /**
262    * Asserts that the value is less than the specified value.
263    *
264    * @param value The values to check against.
265    * @return The response object (for method chaining).
266    * @throws AssertionError If assertion failed.
267    */
268   public R isLessThan(Long value) throws AssertionError {
269      exists();
270      if (! (this.value < value))
271            throw error("Value was not less than expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
272      return returns();
273   }
274
275   /**
276    * Asserts that the value is less than the specified value.
277    *
278    * <p>
279    * Equivalent to {@link #isLessThan(Long)}
280    *
281    * @param value The values to check against.
282    * @return The response object (for method chaining).
283    * @throws AssertionError If assertion failed.
284    */
285   public R isLt(Long value) throws AssertionError {
286      return isLessThan(value);
287   }
288
289   /**
290    * Asserts that the value is less than or equals to the specified value.
291    *
292    * @param value The values to check against.
293    * @return The response object (for method chaining).
294    * @throws AssertionError If assertion failed.
295    */
296   public R isLessThanOrEquals(Long value) throws AssertionError {
297      exists();
298      if (! (this.value <= value))
299            throw error("Value was not less than or equals to expected.\n\tExpected=[{0}]\n\tActual=[{1}]", value, this.value);
300      return returns();
301   }
302
303   /**
304    * Asserts that the value is less than or equals to the specified value.
305    *
306    * <p>
307    * Equivalent to {@link #isLessThanOrEquals(Long)}
308    *
309    * @param value The values to check against.
310    * @return The response object (for method chaining).
311    * @throws AssertionError If assertion failed.
312    */
313   public R isLte(Long value) throws AssertionError {
314      return isLessThanOrEquals(value);
315   }
316
317   /**
318    * Asserts that the value is between (inclusive) the specified upper and lower values.
319    *
320    * @param lower The lower value to check against.
321    * @param upper The upper value to check against.
322    * @return The response object (for method chaining).
323    * @throws AssertionError If assertion failed.
324    */
325   public R isBetween(Long lower, Long upper) throws AssertionError {
326      isLessThanOrEquals(upper);
327      isGreaterThanOrEquals(lower);
328      return returns();
329   }
330
331   /**
332    * Asserts that the value passes the specified predicate test.
333    *
334    * @param test The predicate to use to test the value.
335    * @return The response object (for method chaining).
336    * @throws AssertionError If assertion failed.
337    */
338   public R passes(Predicate<Long> test) throws AssertionError {
339      if (! test.test(value))
340         throw error("Value did not pass predicate test.\n\tValue=[{0}]", value);
341      return returns();
342   }
343
344   // <FluentSetters>
345
346   @Override /* GENERATED - Assertion */
347   public FluentLongAssertion<R> msg(String msg, Object...args) {
348      super.msg(msg, args);
349      return this;
350   }
351
352   @Override /* GENERATED - Assertion */
353   public FluentLongAssertion<R> stderr() {
354      super.stderr();
355      return this;
356   }
357
358   @Override /* GENERATED - Assertion */
359   public FluentLongAssertion<R> stdout() {
360      super.stdout();
361      return this;
362   }
363
364   // </FluentSetters>
