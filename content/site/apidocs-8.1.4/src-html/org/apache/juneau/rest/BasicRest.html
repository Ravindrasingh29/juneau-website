035/**
036 * Identical to {@link BasicRestServlet} but doesn't extend from {@link HttpServlet}.
037 *
038 * <ul class='seealso'>
039 *    <li class='link'>{@doc juneau-rest-server.Instantiation.BasicRest}
040 * </ul>
041 */
042@Rest(
043   // Allow OPTIONS requests to be simulated using ?method=OPTIONS query parameter.
044   allowedMethodParams="OPTIONS"
045)
046@HtmlDocConfig(
047   // Basic page navigation links.
048   navlinks={
049      "up: request:/..",
050      "options: servlet:/?method=OPTIONS",
051      "stats: servlet:/stats"
052   }
053)
054public abstract class BasicRest implements BasicUniversalRest, BasicRestMethods, RestCallHandler, RestInfoProvider, RestCallLogger, RestResourceResolver, ClasspathResourceFinder {
055

057   private volatile RestContext context;
058   private RestCallHandler callHandler;
059   private RestInfoProvider infoProvider;
060   private RestCallLogger callLogger;
061   private ClasspathResourceFinder resourceFinder;
062   private RestResourceResolver resourceResolver = new BasicRestResourceResolver();

209   /**
210    * Method that gets called during servlet initialization.
211    *
212    * <p>
213    * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link RestContextBuilder}
214    * object has been created and initialized with the annotations defined on the class, but before the
215    * {@link RestContext} object has been created.
216    *
217    * <p>
218    * An example of this is the <c>PetStoreResource</c> class that uses an init method to perform initialization
219    * of an internal data structure.
220    *
221    * <h5 class='figure'>Example:</h5>
222    * <p class='bcode w800'>
223    *    <ja>@Rest</ja>(...)
224    *    <jk>public class</jk> PetStoreResource <jk>extends</jk> ResourceJena {
225    *
226    *       <jc>// Our database.</jc>
227    *       <jk>private</jk> Map&lt;Integer,Pet&gt; <jf>petDB</jf>;
228    *
229    *       <ja>@Override</ja>
230    *       <jk>public void</jk> onInit(RestContextBuilder builder) <jk>throws</jk> Exception {
231    *          <jc>// Load our database from a local JSON file.</jc>
232    *          <jf>petDB</jf> = JsonParser.<jsf>DEFAULT</jsf>.parse(getClass().getResourceAsStream(<js>"PetStore.json"</js>), LinkedHashMap.<jk>class</jk>, Integer.<jk>class</jk>, Pet.<jk>class</jk>);
233    *       }
234    *    }
235    * </p>
236    *
237    * <ul class='notes'>
238    *    <li>
239    *       The default implementation of this method is a no-op.
240    *    <li>
241    *       Multiple INIT methods can be defined on a class.
242    *       <br>INIT methods on parent classes are invoked before INIT methods on child classes.
243    *       <br>The order of INIT method invocations within a class is alphabetical, then by parameter count, then by parameter types.
244    *    <li>
245    *       The method can throw any exception causing initialization of the servlet to fail.
246    * </ul>
247    *
248    * @param builder Context builder which can be used to configure the servlet.
249    * @throws Exception Any exception thrown will cause servlet to fail startup.
250    */
251   @RestHook(INIT)
252   public void onInit(RestContextBuilder builder) throws Exception {}
253
254   /**
255    * Method that gets called immediately after servlet initialization.
256    *
257    * <p>
258    * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link RestContext}
259    * object has been created.
260    *
261    * <ul class='notes'>
262    *    <li>
263    *       The default implementation of this method is a no-op.
264    *    <li>
265    *       Multiple POST_INIT methods can be defined on a class.
266    *       <br>POST_INIT methods on parent classes are invoked before POST_INIT methods on child classes.
267    *       <br>The order of POST_INIT method invocations within a class is alphabetical, then by parameter count, then by parameter types.
268    *    <li>
269    *       The method can throw any exception causing initialization of the servlet to fail.
270    * </ul>
271    *
272    * @param context The initialized context object.
273    * @throws Exception Any exception thrown will cause servlet to fail startup.
274    */
275   @RestHook(POST_INIT)
276   public void onPostInit(RestContext context) throws Exception {
277      this.context = context;
278      this.callHandler = new BasicRestCallHandler(context);
279      this.infoProvider = new BasicRestInfoProvider(context);
280      this.callLogger = new BasicRestCallLogger(context);
281      this.resourceFinder = new ClasspathResourceFinderBasic();
282   }

284   /**
285    * Identical to {@link #onPostInit(RestContext)} except the order of execution is child-resources first.
286    *
287    * <p>
288    * Use this method if you need to perform any kind of initialization on child resources before the parent resource.
289    *
290    * <p>
291    * This method is called from within the {@link Servlet#init(ServletConfig)} method after the {@link RestContext}
292    * object has been created and after the {@link HookEvent#POST_INIT} methods have been called.
293    *
294    * <p>
295    * The only valid parameter type for this method is {@link RestContext} which can be used to retrieve information
296    * about the servlet.
297    *
298    * <ul class='notes'>
299    *    <li>
300    *       The default implementation of this method is a no-op.
301    *    <li>
302    *       Multiple POST_INIT_CHILD_FIRST methods can be defined on a class.
303    *       <br>POST_INIT_CHILD_FIRST methods on parent classes are invoked before POST_INIT_CHILD_FIRST methods on child classes.
304    *       <br>The order of POST_INIT_CHILD_FIRST method invocations within a class is alphabetical, then by parameter count, then by parameter types.
305    *    <li>
306    *       The method can throw any exception causing initialization of the servlet to fail.
307    * </ul>
308    *
309    * @param context The initialized context object.
310    * @throws Exception Any exception thrown will cause servlet to fail startup.
311    */
312   @RestHook(POST_INIT_CHILD_FIRST)
313   public void onPostInitChildFirst(RestContext context) throws Exception {}

315   /**
316    * Method that gets called during servlet destroy.
317    *
318    * <p>
319    * This method is called from within the {@link Servlet#destroy()}.
320    *
321    * <h5 class='figure'>Example:</h5>
322    * <p class='bcode w800'>
323    *    <ja>@Rest</ja>(...)
324    *    <jk>public class</jk> PetStoreResource <jk>extends</jk> ResourceJena {
325    *
326    *       <jc>// Our database.</jc>
327    *       <jk>private</jk> Map&lt;Integer,Pet&gt; <jf>petDB</jf>;
328    *
329    *       <ja>@Override</ja>
330    *       <jk>public void</jk> onDestroy(RestContext context) {
331    *          <jf>petDB</jf> = <jk>null</jk>;
332    *       }
333    *    }
334    * </p>
335    *
336    * <ul class='notes'>
337    *    <li>
338    *       The default implementation of this method is a no-op.
339    *    <li>
340    *       Multiple DESTROY methods can be defined on a class.
341    *       <br>DESTROY methods on child classes are invoked before DESTROY methods on parent classes.
342    *       <br>The order of DESTROY method invocations within a class is alphabetical, then by parameter count, then by parameter types.
343    *    <li>
344    *       In general, destroy methods should not throw any exceptions, although if any are thrown, the stack trace will be
345    *       printed to <c>System.err</c>.
346    * </ul>
347    *
348    * @param context The initialized context object.
349    * @throws Exception Any exception thrown will cause stack trace to be printed to <c>System.err</c>.
350    */
351   @RestHook(DESTROY)
352   public void onDestroy(RestContext context) throws Exception {}

354   /**
355    * A method that is called immediately after the <c>HttpServlet.service(HttpServletRequest, HttpServletResponse)</c>
356    * method is called.
357    *
358    * <p>
359    * Note that you only have access to the raw request and response objects at this point.
360    *
361    * <h5 class='figure'>Example:</h5>
362    * <p class='bcode w800'>
363    *    <ja>@Rest</ja>(...)
364    *    <jk>public class</jk> MyResource <jk>extends</jk> BasicRestServlet {
365    *
366    *       <jc>// Add a request attribute to all incoming requests.</jc>
367    *       <ja>@Override</ja>
368    *       <jk>public void</jk> onStartCall(HttpServletRequest req, HttpServletResponse res) {
369    *          req.setAttribute(<js>"foobar"</js>, <jk>new</jk> FooBar());
370    *       }
371    *    }
372    * </p>
373    *
374    * <ul class='notes'>
375    *    <li>
376    *       The default implementation of this method is a no-op.
377    *    <li>
378    *       Multiple START_CALL methods can be defined on a class.
379    *       <br>START_CALL methods on parent classes are invoked before START_CALL methods on child classes.
380    *       <br>The order of START_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
381    *    <li>
382    *       The method can throw any exception.
383    *       <br>{@link HttpException HttpExceptions} can be thrown to cause a particular HTTP error status code.
384    *       <br>All other exceptions cause an HTTP 500 error status code.
385    * </ul>
386    *
387    * @param req The HTTP servlet request object.
388    * @param res The HTTP servlet response object.
389    * @throws Exception Any exception.
390    */
391   @RestHook(START_CALL)
392   public void onStartCall(HttpServletRequest req, HttpServletResponse res) throws Exception {}

394   /**
395    * Method that gets called immediately before the <ja>@RestMethod</ja> annotated method gets called.
396    *
397    * <p>
398    * At this point, the {@link RestRequest} object has been fully initialized, and all {@link RestGuard} and
399    * {@link RestMatcher} objects have been called.
400    *
401    * <ul class='notes'>
402    *    <li>
403    *       The default implementation of this method is a no-op.
404    *    <li>
405    *       Multiple PRE_CALL methods can be defined on a class.
406    *       <br>PRE_CALL methods on parent classes are invoked before PRE_CALL methods on child classes.
407    *       <br>The order of PRE_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
408    *    <li>
409    *       The method can throw any exception.
410    *       <br>{@link HttpException HttpExceptions} can be thrown to cause a particular HTTP error status code.
411    *       <br>All other exceptions cause an HTTP 500 error status code.
412    *    <li>
413    *       It's advisable not to mess around with the HTTP body itself since you may end up consuming the body
414    *       before the actual REST method has a chance to use it.
415    * </ul>
416    *
417    * @param req The request object.
418    * @param res The response object.
419    * @throws Exception Any exception.
420    */
421   @RestHook(PRE_CALL)
422   public void onPreCall(RestRequest req, RestResponse res) throws Exception {}

424   /**
425    * Method that gets called immediately after the <ja>@RestMethod</ja> annotated method gets called.
426    *
427    * <p>
428    * At this point, the output object returned by the method call has been set on the response, but
429    * {@link RestConverter RestConverters} have not yet been executed and the response has not yet been written.
430    *
431    * <ul class='notes'>
432    *    <li>
433    *       The default implementation of this method is a no-op.
434    *    <li>
435    *       Multiple POST_CALL methods can be defined on a class.
436    *       <br>POST_CALL methods on parent classes are invoked before POST_CALL methods on child classes.
437    *       <br>The order of POST_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
438    *    <li>
439    *       The method can throw any exception, although at this point it is too late to set an HTTP error status code.
440    * </ul>
441    *
442    * @param req The request object.
443    * @param res The response object.
444    * @throws Exception Any exception.
445    */
446   @RestHook(POST_CALL)
447   public void onPostCall(RestRequest req, RestResponse res) throws Exception {}

449   /**
450    * Method that gets called right before we exit the servlet service method.
451    *
452    * <p>
453    * At this point, the output has been written and flushed.
454    *
455    * <p>
456    * The following attributes are set on the {@link HttpServletRequest} object that can be useful for logging purposes:
457    * <ul>
458    *    <li><js>"Exception"</js> - Any exceptions thrown during the request.
459    *    <li><js>"ExecTime"</js> - Execution time of the request.
460    * </ul>
461    *
462    * <ul class='notes'>
463    *    <li>
464    *       The default implementation of this method is a no-op.
465    *    <li>
466    *       Multiple END_CALL methods can be defined on a class.
467    *       <br>END_CALL methods on parent classes are invoked before END_CALL methods on child classes.
468    *       <br>The order of END_CALL method invocations within a class is alphabetical, then by parameter count, then by parameter types.
469    *    <li>
470    *       The method can throw any exception, although at this point it is too late to set an HTTP error status code.
471    *    <li>
472    *       Note that if you override a parent method, you probably need to call <code><jk>super</jk>.parentMethod(...)</code>.
473    *       <br>The method is still considered part of the parent class for ordering purposes even though it's
474    *       overridden by the child class.
475    * </ul>
476    *
477    * @param req The HTTP servlet request object.
478    * @param res The HTTP servlet response object.
479    * @throws Exception Any exception.
480    */
481   @RestHook(END_CALL)
482   public void onEndCall(HttpServletRequest req, HttpServletResponse res) throws Exception {}
