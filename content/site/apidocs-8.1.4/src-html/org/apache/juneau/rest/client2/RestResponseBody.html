210   /**
211    * Returns the HTTP response message body as an input stream.
212    *
213    * <ul class='notes'>
214    *    <li>
215    *       Once this input stream is exhausted, it will automatically be closed.
216    *  <li>
217    *    This method can be called multiple times if {@link #cache()} has been called.
218    *  <li>
219    *    Calling this method multiple times without caching enabled will cause a {@link RestCallException}
220    *    with an inner {@link IllegalStateException} to be thrown.
221    * </ul>
222    *
223    * @return
224    *    The HTTP response message body input stream, never <jk>null</jk>.
225    *    <br>For responses without a body(e.g. HTTP 204), returns an empty stream.
226    * @throws RestCallException If a stream or illegal state exception was thrown.
227    */
228   public InputStream asInputStream() throws RestCallException {
229      try {
230         if (cache != null)
231            return new ByteArrayInputStream(cache);
232
233         if (cached) {
234            cache = IOUtils.readBytes(entity.getContent());
235            response.close();
236            return new ByteArrayInputStream(cache);
237         }
238
239         if (isConsumed && ! entity.isRepeatable())
240            throw new IllegalStateException("Method cannot be called.  Response has already been consumed.");
241
242         HttpEntity e = response.asHttpResponse().getEntity();
243         InputStream is = e == null ? new ByteArrayInputStream(new byte[0]) : e.getContent();
244
245         is = new EofSensorInputStream(is, new EofSensorWatcher() {
246            @Override
247            public boolean eofDetected(InputStream wrapped) throws IOException {
248               response.close();
249               return true;
250            }
251            @Override
252            public boolean streamClosed(InputStream wrapped) throws IOException {
253               response.close();
254               return true;
255            }
256            @Override
257            public boolean streamAbort(InputStream wrapped) throws IOException {
258               response.close();
259               return true;
260            }
261         });
262
263         isConsumed = true;
264
265         return is;
266      } catch (UnsupportedOperationException | IOException e) {
267         throw new RestCallException(e);
268      }
269   }

271   /**
272    * Returns the HTTP response message body as a reader based on the charset on the <code>Content-Type</code> response header.
273    *
274    * <ul class='notes'>
275    *    <li>
276    *       If no charset was found on the <code>Content-Type</code> response header, <js>"UTF-8"</js> is assumed.
277    *    <li>
278    *       Once this input stream is exhausted, it will automatically be closed.
279    *  <li>
280    *    This method can be called multiple times if {@link #cache()} has been called.
281    *  <li>
282    *    Calling this method multiple times without caching enabled will cause a {@link RestCallException}
283    *    with an inner {@link IllegalStateException} to be thrown.
284    * </ul>
285    *
286    * @return
287    *    The HTTP response message body reader, never <jk>null</jk>.
288    *    <br>For responses without a body(e.g. HTTP 204), returns an empty reader.
289    * @throws RestCallException If an exception occurred.
290    */
291   public Reader asReader() throws RestCallException {
292
293      // Figure out what the charset of the response is.
294      String cs = null;
295      String ct = getContentType().asString();
296
297      // First look for "charset=" in Content-Type header of response.
298      if (ct != null && ct.contains("charset="))
299         cs = ct.substring(ct.indexOf("charset=")+8).trim();
300
301      return asReader(cs);
302   }

474   /**
475    * Parses HTTP body into the specified object type.
476    *
477    * <p>
478    * The type can be a simple type (e.g. beans, strings, numbers) or parameterized type (collections/maps).
479    *
480    * <h5 class='section'>Examples:</h5>
481    * <p class='bcode w800'>
482    *    <jc>// Parse into a linked-list of strings.</jc>
483    *    List&lt;String&gt; l = client.get(<jsf>URL</jsf>).run()
484    *       .getBody().as(LinkedList.<jk>class</jk>, String.<jk>class</jk>);
485    *
486    *    <jc>// Parse into a linked-list of beans.</jc>
487    *    List&lt;MyBean&gt; l = client.get(<jsf>URL</jsf>).run()
488    *       .getBody().as(LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
489    *
490    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
491    *    List&lt;List&lt;String&gt;&gt; l = client.get(<jsf>URL</jsf>).run()
492    *       .getBody().as(LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
493    *
494    *    <jc>// Parse into a map of string keys/values.</jc>
495    *    Map&lt;String,String&gt; m = client.get(<jsf>URL</jsf>).run()
496    *       .getBody().as(TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
497    *
498    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
499    *    Map&lt;String,List&lt;MyBean&gt;&gt; m = client.get(<jsf>URL</jsf>).run()
500    *       .getBody().as(TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
501    * </p>
502    *
503    * <p>
504    * <c>Collection</c> classes are assumed to be followed by zero or one objects indicating the element type.
505    *
506    * <p>
507    * <c>Map</c> classes are assumed to be followed by zero or two meta objects indicating the key and value types.
508    *
509    * <p>
510    * The array can be arbitrarily long to indicate arbitrarily complex data structures.
511    *
512    * <ul class='notes'>
513    *    <li>
514    *       Use the {@link #as(Class)} method instead if you don't need a parameterized map/collection.
515    *    <li>
516    *       You can also specify any of the following types:
517    *       <ul>
518    *          <li class='jc'>{@link RestResponseBody}/{@link HttpEntity} - Returns access to this object.
519    *          <li class='jc'>{@link Reader} - Returns access to the raw reader of the response.
520    *          <li class='jc'>{@link InputStream} - Returns access to the raw input stream of the response.
521    *          <li class='jc'>{@link ReaderResource} - Returns access as a reader wrapped in a reader resource.
522    *          <li class='jc'>{@link StreamResource} - Returns access as an input stream wrapped in a stream resource.
523    *       </ul>
524    *    <li>
525    *    If {@link #cache()} has been called, this method can be can be called multiple times and/or combined with
526    *    other methods that retrieve the content of the response.  Otherwise a {@link RestCallException}
527    *    with an inner {@link IllegalStateException} will be thrown.
528    *    <li>
529    *       The input stream is automatically closed after this call.
530    * </ul>
531    *
532    * @param <T> The class type of the object to create.
533    * @param type
534    *    The object type to create.
535    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
536    * @param args
537    *    The type arguments of the class if it's a collection or map.
538    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
539    *    <br>Ignored if the main type is not a map or collection.
540    * @return The parsed object.
541    * @throws RestCallException
542    *    <ul>
543    *       <li>If the input contains a syntax error or is malformed, or is not valid for the specified type.
544    *       <li>If a connection error occurred.
545    *    </ul>
546    * @see BeanSession#getClassMeta(Class) for argument syntax for maps and collections.
547    */
548   public <T> T as(Type type, Type...args) throws RestCallException {
549      return as(getClassMeta(type, args));
550   }

595      m.set(as(type, args));
596      return response;

601    *
602    * <p>

727    *    other methods that retrieve the content of the response.  Otherwise a {@link RestCallException}
728    *    with an inner {@link IllegalStateException} will be thrown.
729    *    <li>
730    *       The input stream is automatically closed after this call.
731    * </ul>
732    *
733    * @param <T> The class type of the object to create.
734    * @param type The object type to create.
735    * @return The parsed object.
736    * @throws RestCallException
737    *    <ul>
738    *       <li>If the input contains a syntax error or is malformed, or is not valid for the specified type.
739    *       <li>If a connection error occurred.
740    *    </ul>
741    * @see BeanSession#getClassMeta(Class) for argument syntax for maps and collections.
742    */
743   @SuppressWarnings("unchecked")
744   public <T> T as(ClassMeta<T> type) throws RestCallException {
745      try {
746         Class<?> ic = type.getInnerClass();
747
748         if (ic.equals(RestResponseBody.class) || ic.equals(HttpEntity.class))
749            return (T)this;
750
751         if (ic.equals(Reader.class))
752            return (T)asReader();
753
754         if (ic.equals(InputStream.class))
755            return (T)asInputStream();
756
757         if (type.isType(HttpResponse.class))
758            return (T)response;
759
760         if (type.isType(ReaderResource.class) || type.isType(StreamResource.class)) {
761            String mediaType = null;
762            ObjectMap headers = new ObjectMap();
763            for (Header h : response.getAllHeaders()) {
764               if (h.getName().equalsIgnoreCase("Content-Type"))
765                  mediaType = h.getValue();
766               else
767                  headers.put(h.getName(), h.getValue());
768            }
769            if (type.isType(ReaderResource.class))
770               return (T)ReaderResource.create().headers(headers).mediaType(mediaType).contents(asReader()).build();
771            return (T)StreamResource.create().headers(headers).mediaType(mediaType).contents(asInputStream()).build();
772         }
773
774         String ct = firstNonEmpty(response.getHeader("Content-Type").asStringOrElse("text/plain"));
775
776         MediaType mt = MediaType.forString(ct);
777
778         if (parser == null || (mt.toString().equals("text/plain") && ! parser.canHandle(ct))) {
779            if (type.hasStringMutater())
780               return type.getStringMutater().mutate(asString());
781         }
782
783         if (parser != null) {
784            try (Closeable in = parser.isReaderParser() ? asReader() : asInputStream()) {
785
786               ParserSessionArgs pArgs =
787                  ParserSessionArgs
788                     .create()
789                     .properties(new ObjectMap().setInner(request.getProperties()))
790                     .locale(response.getLocale())
791                     .mediaType(mt)
792                     .schema(schema);
793
794               T t = parser.createSession(pArgs).parse(in, type);
795
796               // Some HTTP responses have no body, so try to create these beans if they've got no-arg constructors.
797               if (t == null && ! type.isType(String.class)) {
798                  ConstructorInfo c = type.getInfo().getPublicConstructor();
799                  if (c != null) {
800                     try {
801                        return c.<T>invoke();
802                     } catch (ExecutableException e) {
803                        throw new ParseException(e);
804                     }
805                  }
806               }
807
808               return t;
809            }
810         }
811
812         if (type.hasReaderMutater())
813            return type.getReaderMutater().mutate(asReader());
814
815         if (type.hasInputStreamMutater())
816            return type.getInputStreamMutater().mutate(asInputStream());
817
818         throw new ParseException(
819            "Unsupported media-type in request header ''Content-Type'': ''{0}''\n\tSupported media-types: {1}",
820            response.getStringHeader("Content-Type"), parser == null ? null : parser.getMediaTypes()
821         );
822
823      } catch (ParseException | IOException e) {
824         response.close();
825         throw new RestCallException(e);
826      }
827   }

1707    *    <li>
1708    *       The input stream is automatically closed after this call.
1709    * </ul>
1710    *
1711    * @param pattern The pattern to test for.
1712    * @return The response object (for method chaining).
1713    * @throws RestCallException If assertion fails.
1714    */
1715   public RestResponse assertValueMatches(Pattern pattern) throws RestCallException {
1716      String text = asString();
1717      if (! pattern.matcher(text).matches())
1718         throw new RestCallException("Response did not match expected pattern.\n\tpattern=[{0}]\n\tBody=[{1}]", pattern.pattern(), text);
1719      return response;
1720   }
