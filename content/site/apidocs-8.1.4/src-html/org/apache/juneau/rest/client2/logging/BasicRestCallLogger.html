013package org.apache.juneau.rest.client2.logging;
014
015import java.util.logging.*;
016
017import org.apache.juneau.rest.client2.*;
018
019/**
020 * Specialized interceptor for logging calls to a log file.
021 *
022 * <p>
023 * Causes a log entry to be created that shows all the request and response headers and content at the end of the
024 * request.
025 *
026 * <p>
027 * Use the {@link RestClientBuilder#logTo(Level, Logger)} and {@link RestRequest#logTo(Level, Logger)} methods to create
028 * instances of this class.
029 */
030public class BasicRestCallLogger extends RestCallLogger {
031
032   /**
033    * Default HTTP request logger.
034    * <p>
035    * Logs outgoing HTTP requests to the <c>org.apache.juneau.rest.client</c> logger at <jsf>WARNING</jsf> level.
036    */
037   public static final BasicRestCallLogger DEFAULT = new BasicRestCallLogger(Level.WARNING, Logger.getLogger("org.apache.juneau.rest.client"));
038
039   private Level level;
040   private Logger log;
041
042   /**
043    * Constructor.
044    *
045    * @param level The log level to log messages at.
046    * @param log The logger to log to.
047    */
048   public BasicRestCallLogger(Level level, Logger log) {
049      this.level = level;
050      this.log = log;
051   }
052
053   @Override
054   public boolean shouldLog(RestRequest req, RestResponse res) {
055      return log.isLoggable(level);
056   }
057
058   @Override
059   public void log(Throwable t, String msg) {
060      log.log(level, msg, t);
061   }
062}
